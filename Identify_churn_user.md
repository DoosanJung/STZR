    import pytz


    def utc_to_each_tz(utc_dt, region):
        import pytz
        local_tz = pytz.timezone(region)
        local_dt = utc_dt.replace(tzinfo=pytz.utc).astimezone(local_tz)
        return local_tz.normalize(local_dt)
